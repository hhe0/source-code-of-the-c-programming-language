#### 编写trim函数：删除字符串尾部的空格符、制表符与换行符。

```C
#include <stdio.h>
#include <string.h>

void trim(char s[]);

int main()
{
	char s[] = "hhe-fdsaf   ";
	
	trim(s);
	
	printf("%s\n", s);
}

void trim(char s[])
{
	int len = strlen(s);
	int i;
	
	for (i = len-1; i >= 0; i--)
	{
		if (s[i] == ' ' || s[i] == '\t' || s[i] == '\n')
		{
			s[i] = '\0';
		}
	}
}
```
