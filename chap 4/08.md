#### 给计算器程序增加访问sin、exp与pow等库函数的操作。

* 随着操作符的增多，尤其是多个字符组成的操作符，现有程序的扩充出现了一些困难，待优化。

```C
#include <stdio.h>
#include <string.h>
#include <ctype.h>
#include <math.h>

#define MAXSIZE 20

// 使用一个公共的数组用来模拟栈，外部变量存在默认值
// 这样做的好处在于不需要在函数中做无意义的参数传递
static int stack[MAXSIZE];
// 维护一个栈顶指针
static int ptr = 0;

void calculator(char s[]);
void push(int num);
int pop();
void printArr();
int isoperator(char c);

int main()
{
	char s[] = "1 -5 + 2 * 10 + 9 * 3 / -1 * -4 % 3 + e s";
	int len = strlen(s);
	int i;
	int validate = 0;
	
	for (i = len-1; i >= 0; i--)
	{
	    if (s[i] == ' ')
	    {
	        continue;
	    }
	    else 
	    {
	        if (!isoperator(s[i]))
	        {
	            break;
	        }
	        else 
	        {
	            validate = 1;
	            break;
	        }
	    }
	}
	
	if (validate)
	{
	    calculator(s);
	
	    printf("%d\n", pop());
	}
	else
	{
	    printf("Unavailable input! Please check!");
	}

	
	return 0;
}

void calculator(char s[])
{
	int len = strlen(s);
	int i;
	int op_a;
	int op_b;
	int num;
	int signal;
	
	for (i = 0; i < len; i++)
	{
	    printArr();
	    printf("%c : \n", s[i]);
	    
		switch (s[i])
		{
			case ' ':
				break;
			case '+' :
				push(pop() + pop());	
				break;
			case '-' :
			    // 判断当前是减号还是负号
			    if (' ' != s[i+1])
			    {
			        break;
			    }
				op_a = pop();
				op_b = pop();
				push(op_b - op_a);
				break;
			case '*' :
				push(pop() * pop());
				break;
			case '/' :
				op_a = pop();
				op_b = pop();
				push(op_b / op_a);
				break;
			case '%' :
				// 注意取模运算的结果符号与被除数相同
				op_a = pop();
				op_b = pop();
				push(op_b % op_a);
				break;
			case 's' :
				// sin运算
				push(sin(pop()));
				break;
			case 'e':
				// exp运算
				push(exp(pop()));
				break;
			case 'p':
				// power运算
				op_a = pop();
				op_b = pop();
				push(pow(op_b, op_a));
				break;
			default:
			    num = 0;
			    signal = 1;
			    
				// 为数字的情况
				if (s[i-1] == '-')
				{
				    signal = -1;
				}
				do 
				{
					num = num * 10 + (s[i] - '0');
					i++;
				}
				while (' ' != s[i]);
				push(num * signal);
				i--;
				break;
		}
	}
}

// push进去的都是处理后的数字
void push(int num)
{
	stack[ptr] = num;
	
	ptr++;
}

// 弹出栈顶值
int pop()
{
	ptr--;
	
	return stack[ptr];
}

//查看栈数据的变化
void printArr()
{
    int i;
    
    for (i = 0; i < MAXSIZE; i++)
    {
        printf("%d ", stack[i]);
    }
    
    printf("\n");
}

// 是否为操作符
int isoperator(char c)
{
	return (c == '+') || (c == '-') | (c == '*') || (c == '/') || (c == '%') || (c == 's') || (c == 'e') || (c == 'p');
}
```
