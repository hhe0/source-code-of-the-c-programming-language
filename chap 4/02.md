#### 编写函数strindex(s, t)，它返回字符串t在s中最右边出现的位置。如果s中不包含t，则返回-1。

```C
#include <stdio.h>
#include <string.h>

int strindex(char s[], char t[]);

int main()
{
	char s[] = "abchheabchhhe";
	char t[] = "hhe";
	
	int pos = strindex(s, t);
	
	printf("%d\n", pos);

	return 0 ;
}

int strindex(char s[], char t[])
{
	int s_len = strlen(s);
	int t_len = strlen(t);
	int i;
	int j;
	int found = 0;
	
	for (i = s_len - 1; i >= t_len - 1; i--)
	{
		for (j = t_len - 1; j >= 0; j--)
		{
			if (s[i-((t_len-1)-j)] != t[j])
			{
				break;
			}
		}
		if (j < 0)
		{
			return i-(t_len-1);
		}
	}
	
	return -1;
}
```
