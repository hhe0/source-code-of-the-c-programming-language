#### 对atof函数进行扩充，使它可以处理形如123.45e-6的科学表示法，其中，浮点数后面可能会紧跟一个E或者e以及一个指数（可能有正负号）。

* 简易版本的函数，后续再慢慢改进。

```C
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <ctype.h>

int isDigit(char c);
double atof(char s[]);

int main()
{
	char s[] = "123.1221e12";
	
	double num = atof(s);
	
	printf("%lf\n", num);

	return 0;
}

double atof(char s[])
{
	double num_int = 0;
	double num_decimal = 0;
	double num_index = 0;
	int s_len = strlen(s);
	int pos = 0;
	int pos_dot = s_len;
	int pos_e = s_len;
	
	while (pos < s_len)
	{
		if ('.' == s[pos])
		{
			// 找到小数点所在的位置
			pos_dot = pos;
		}
		else if ('e' == tolower(s[pos]))
		{
			// 找到科学计数法e的位置
			pos_e = pos;
			break;
		}
		
		pos++;
	}
	
	// 处理整数部分，可以兼容无小数部分的情况
	int i;
	for (i = 0; i < pos_dot; i++)
	{
		if (isDigit(s[i]))
		{
			num_int = num_int * 10 + (s[i] - '0');
		}
	}
	
	// 处理小数部分
	// 123.45e-6
	int j;
	
	if (pos_dot < s_len) 
	{
		for (j = pos_e - 1; j > pos_dot; j--)
		{
			if (isDigit(s[j]))
			{
				num_decimal = num_decimal * 0.1 + (s[j] - '0') * 0.1;
			}
		}
	}
	
	// 处理科学计数法e后面的部分
	int k;
	int signal = 1;
	
	if (pos_e < s_len)
	{
		if ('-' == s[pos_e+1])
		{
			signal = -1;
			// 为了兼容for循环的pos_e+1
			pos_e++;
		}
		for (k = pos_e + 1; k < s_len; k++)
		{
			if (isDigit(s[k]))
			{
				num_index = num_index * 10 + (s[k] - '0');
			}
		}
	}
	
	printf("%lf %lf %lf %d\n", num_int, num_decimal, num_index, signal);
	
	// 注意是pow函数，而不是power函数
	return (num_int + num_decimal) * pow(10, num_index * signal);
}

int isDigit(char c)
{
	return (c >= '0' && c <= '9');
}
```
