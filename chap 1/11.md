#### 编写函数reverse(s)，将字符串s中的字符顺序颠倒过来。使用该函数编写一个程序，每次颠倒一个输入行中的字符顺序。
* 解法一
```C
#include <stdio.h>

#define MAXLINE 1000

int getline(char line[], int maxline);
void reverse(char s[]);

int main()
{
    int c;
    char line[MAXLINE];

    while (getline(line, MAXLINE)) {
        reverse(line);
        printf("%s\n", line);
    }
}

int getline(char s[], int lim)
{
    int c, i;
    int j = 0;

    for (i = 0; (c = getchar()) != EOF && c != '\n'; i++) {
        if (i < lim - 2) {
            s[i] = c;
            j++;
        }

        if(c == '\n') {
            s[i] = c;
            i++;
            j++;
        }

        s[j] = '\0';
    }

    return i;
}

// 非常好的算法
void reverse(char s[])
{
    int i = 0;
    int j = 0;
    int temp;

    while (s[i] != '\0') {
        ++i;
    }

    i--;

    if (s[i] == '\n') {
        i--;
    }

    while (j < i) {
        temp = s[j];
        s[j] = s[i];
        s[i] = temp;
        i--;
        j++;
    }
}
```

* 解法二
```C
#include <stdio.h>
#include <string.h>

#define MAX_SIZE 1000
#define MAX_LINE

int getline(char s[], int lim);
void reverse(char s[]);

int main()
{
    char s[MAX_SIZE];
    int len;
    
    while ((len = getline(s, MAX_SIZE)) > 0) {
        reverse(s);
        printf("%s", s);
    }
}

int getline(char s[], int lim)
{
    int c, i;

    for (i = 0; i < lim-1 && (c = getchar()) != EOF && c != '\n'; i++) {
        s[i] = c;
    }

    // 考虑到可能为单行的情况
    // 这里的处理是将换行符也视为一个字符
    if(c == '\n') {
        s[i] = c;
        i++;
    }

    s[i] = '\0';

    return i;
}

void reverse(char s[]) 
{
    int len = strlen(s);
    int i = 0;
    int j = len - 1;
    char temp;
    
   while (i < len && i <= j) {
       if (s[j] == '\n') {
           j--;
           continue;
       }
       else {
           temp = s[i];
           s[i] = s[j];
           s[j] = temp;
           i++;
           j--;
       }
   }
}
```
